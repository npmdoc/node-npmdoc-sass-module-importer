    function index(opts) {
  var importer = new ModuleImporter(opts);

  return function (url, prev, done) {
    importer.resolve({ url: url, prev: prev })
      .then(done)
      .catch(function ( err ) { return setImmediate(function () { throw err; }); });
  };
}
